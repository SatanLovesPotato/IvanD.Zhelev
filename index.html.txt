        function switchLanguage(lang) {
            const elements = document.querySelectorAll('[lang]');
            elements.forEach(element => {
                if (element.getAttribute('lang') === lang) {
                    element.style.display = 'block';
                } else {
                    element.style.display = 'none';
                }
            });
        }
